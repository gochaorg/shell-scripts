1 /**
  2 @class Объект для работы с сетью
  3 */
  4 var net = {
  5 	_net : null,
  6 	_init : function(){
  7 		if( this._net==null )this._net = WScript.CreateObject("WScript.Network");
  8 	},
  9 	/**
 10 	Возвращает имя компьютера
 11 	@returns {string} Имя компьютера
 12 	*/
 13 	computerName : function(){
 14 		this._init();
 15 		return this._net.ComputerName;
 16 	},
 17 	/**
 18 	Возвращает имя пользователя в домене
 19 	@returns {string} Имя пользователя в домене
 20 	*/
 21 	userDomain : function(){
 22 		this._init();
 23 		return this._net.UserDomain;
 24 	},
 25 	/**
 26 	Возвращает имя пользователя
 27 	@returns {string} Имя пользователя
 28 	*/
 29 	userName : function(){
 30 		this._init();
 31 		return this._net.UserName;
 32 	},
 33 	/**
 34 	Добавить виндовый сетевой принтер
 35 	@param {string} printerPath Путь к принтеру. Например: <code>"\\\\Server\\Print1"</code>
 36 	@param {string} driver=optional Имя драйвера. Используется в Windows 9x. Параметр не обязательный.
 37 	@param {string} port=optional порт (LPT1). Используется в Windows 9x. Параметр не обязательный.
 38 	*/
 39 	addWindowsPrinter : function(printerPath){
 40 		this._init();
 41 		if( arguments.length==3 )return this._net.AddWindowsPrinterConnection(printerPath,arguments[1],arguments[2]);
 42 		if( arguments.length==2 )return this._net.AddWindowsPrinterConnection(printerPath,arguments[1]);
 43 		return this._net.AddWindowsPrinterConnection(printerPath);
 44 	},
 45 	
 46 	/*
 47 	Добавляет сетевой принтер MS-DOS к вашему компьютеру.
 48 	@param {string} localName Локальное название принтера
 49 	@param {string} printerPath Сетевой путь к принтеру
 50 	@param {bool} updateProfile=optional (true/false) Обновлять профиль пользователя
 51 	@param {string} userName=optional Логин пользователя
 52 	@param {string} password=optional Пароль пользователя
 53 	addPrinter : function(localName,printerPath,updateProfile,userName,password){
 54 		this._init();
 55 		if( arguments.length==2 )
 56 			return this._net.AddPrinterConnection( localName, printerPath );
 57 		if( arguments.length==3 )
 58 			return this._net.AddPrinterConnection( localName, printerPath, updateProfile );
 59 		if( arguments.length==4 )
 60 			return this._net.AddPrinterConnection( localName, printerPath, updateProfile, userName );
 61 		if( arguments.length==5 )
 62 			return this._net.AddPrinterConnection( localName, printerPath, updateProfile, userName, password );
 63 	},
 64 	*/
 65 	
 66 	/**
 67 	Монтирует сетевой диск
 68 	@param {string} drive - Имя диска. Например <b>"E:"</b>
 69 	@param {string} smbPath - Сетевой путь. Например <b>"\\\\Server\\Public"</b>
 70 	@param {bool} updateProfile=optional (true/false) обновлять профиль пользователя
 71 	@param {string} userName=optional Логин пользователя на сервере
 72 	@param {string} password=optional Пароль пользователя на сервере
 73 	*/
 74 	mountSmb : function(drive,smbPath,updateProfile,userName,password){
 75 		this._init();
 76 		if( arguments.length>0 ){
 77 			if( typeof(drive)=="string" ){
 78 				if( drive.length==1 )drive = drive + ":";
 79 				else if( drive.length==3 )drive = drive.substr(0,2);
 80 			}
 81 		}
 82 		if( arguments.length==2 )return this._net.MapNetworkDrive( drive, smbPath );
 83 		if( arguments.length==3 )return this._net.MapNetworkDrive( drive, smbPath,updateProfile );
 84 		if( arguments.length==4 )return this._net.MapNetworkDrive( drive, smbPath,updateProfile,userName );
 85 		if( arguments.length==5 )return this._net.MapNetworkDrive( drive, smbPath,updateProfile,userName,password );
 86 	},
 87 	
 88 	/**
 89 	Демонтирует сетевой диск
 90 	@param {string} drive - Имя диска. Например <b>"E:"</b>
 91 	@param {bool} force=optional (true/false) Форсировать, даже если диск используется
 92 	@param {bool} updateProfile=optional (true/false) обновлять профиль пользователя
 93 	*/
 94 	umountSmb : function( drive,force,updateProfile ){
 95 		this._init();
 96 		if( arguments.length>0 ){
 97 			if( typeof(drive)=="string" ){
 98 				if( drive.length==1 )drive = drive + ":";
 99 				else if( drive.length==3 )drive = drive.substr(0,2);
100 			}
101 		}
102 		if( arguments.length==1 )return this._net.RemoveNetworkDrive( drive );
103 		if( arguments.length==2 )return this._net.RemoveNetworkDrive( drive,force );
104 		if( arguments.length==3 )return this._net.RemoveNetworkDrive( drive,force,updateProfile );
105 	},
106 	
107 	/**
108 	Возвращает ассоциативный массив смонтированных сетевых дисков
109 	@returns {array[string/string]} ассоциативный массив смонтированных сетевых дисков: Диск / Сетевой путь
110 	*/
111 	mountedSmb : function(){
112 		this._init();
113 		var drives = this._net.EnumNetworkDrives();
114 		var res = [];
115 		for(var i = 0; i < drives.length; i += 2) {
116 			res[drives.Item(i)] = drives.Item(i + 1);
117         }
118 		return res;
119 	}
120 };
