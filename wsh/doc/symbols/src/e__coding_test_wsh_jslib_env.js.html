1 /**
  2 @class Работа с опкружением
  3 */
  4 var env = {
  5 	/**
  6 	Возвращает массив переданных аргументов.
  7 	@returns {string[]} массив переданных аргументов.
  8 	*/
  9 	args : function() {
 10 		var res = [];
 11 		var objArgs = WScript.Arguments;
 12 		for (var i = 0; i < objArgs.length; i++)
 13 		{
 14 		   res[res.length] = objArgs(i);
 15 		}
 16 		return res;
 17 	},
 18 	
 19 	_shell : null,
 20 	
 21 	_init : function(){
 22 		if( this._shell==null )this._shell = WScript.CreateObject("WScript.Shell");
 23 	},
 24 	
 25 	/**
 26 	Возвращает ассоц-массив переменных окружения
 27 	@returns {array[string/string]} Ассоц. массив ключ / значение переменных окружения.
 28 	*/
 29 	vars : function() {
 30 		this._init();
 31 		var res = [];
 32 		var en = new Enumerator(this._shell.Environment);
 33 		for( ; !en.atEnd() ; en.moveNext() ){
 34 			var k = en.item();
 35 			var key = "";
 36 			for( var i=0; i<k.length; i++ ){
 37 				var c = k.charAt(i);
 38 				if( c=='=' )break;
 39 				key = key + c;
 40 			}
 41 			var val = this._shell.Environment.Item( key );
 42 			res[key] = val;
 43 		}
 44 		return res;
 45 	},
 46 	
 47 	/**
 48 	Возвращает значение специальной папки
 49 	@param {string} name Имя папки.
 50 	Возможные значения:
 51 	<ul>
 52 		<li>AllUsersDesktop</li>
 53 		<li>AllUsersStartMenu</li>
 54 		<li>AllUsersPrograms</li>
 55 		<li>Desktop</li>
 56 		<li>Favorites</li>
 57 		<li>Fonts</li>
 58 		<li>MyDocuments</li>
 59 		<li>NetHood</li>
 60 		<li>PrintHood</li>
 61 		<li>Programs</li>
 62 		<li>Recent</li>
 63 		<li>SendTo</li>
 64 		<li>StartMenu</li>
 65 		<li>Startup</li>
 66 		<li>Templates</li>
 67 	</ul>
 68 	@returns {string} Значение
 69 	*/
 70 	specialFolder : function( name ) {
 71 		this._init();
 72 		return this._shell.SpecialFolders( name );
 73 	},
 74 	
 75 	/**
 76 	Возвращает ассоциативный массив специальных папок.
 77 	@returns {array[string/string]} ассоциативный массив специальных папок.
 78 	*/
 79 	specialFolders : function(){
 80 		var res = [];
 81 		res["AllUsersDesktop"] = this.specialFolder( "AllUsersDesktop" );
 82 		res["AllUsersStartMenu"] = this.specialFolder( "AllUsersStartMenu" );
 83 		res["AllUsersPrograms"] = this.specialFolder( "AllUsersPrograms" );
 84 		res["Desktop"] = this.specialFolder( "Desktop" );
 85 		res["Favorites"] = this.specialFolder( "Favorites" );
 86 		res["Fonts"] = this.specialFolder( "Fonts" );
 87 		res["MyDocuments"] = this.specialFolder( "MyDocuments" );
 88 		res["NetHood"] = this.specialFolder( "NetHood" );
 89 		res["PrintHood"] = this.specialFolder( "PrintHood" );
 90 		res["Programs"] = this.specialFolder( "Programs" );
 91 		res["Recent"] = this.specialFolder( "Recent" );
 92 		res["SendTo"] = this.specialFolder( "SendTo" );
 93 		res["StartMenu"] = this.specialFolder( "StartMenu" );
 94 		res["Startup"] = this.specialFolder( "Startup" );
 95 		res["Templates"] = this.specialFolder( "Templates" );
 96 		return res;
 97 	}
 98 };
